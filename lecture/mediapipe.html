MediaPipe - Hands 기초: 손 랜드마크

2. 손 랜드마크 검출의 핵심

MediaPipe Hands는 각 손에서 21개의 3D 랜드마크를 실시간으로 추출합니다. 손목부터 손가락 끝까지의 정확한 위치를 픽셀 좌표로 얻을 수 있습니다.

RGB 변환 이유: MediaPipe 모델은 RGB 입력을 기대하지만 OpenCV는 BGR을 사용하므로 변환이 필수입니다.

성능 최적화: writeable=False 설정으로 메모리 복사를 방지해 처리 속도를 향상시킵니다.

🚀 실행 및 조정

python mediapipe_5_webcam.py

손이 잘 인식되지 않으면 min_detection_confidence를 0.3으로 낮추세요

import cv2, mediapipe as mp

# MediaPipe 솔루션 초기화
mp_hands = mp.solutions.hands
mp_draw = mp.solutions.drawing_utils
cap = cv2.VideoCapture(0)

# 손 검출 설정: 최대 2개 손, 신뢰도 임계값
with mp_hands.Hands(
    max_num_hands=2,                    # 동시 추적 가능한 손 개수
    min_detection_confidence=0.5,      # 최초 검출 신뢰도
    min_tracking_confidence=0.5       # 추적 유지 신뢰도
) as hands:
    
  while cap.isOpened():
    ret, frame = cap.read()
    if not ret: break
    
    # BGR→RGB 변환 (MediaPipe 요구사항)
    rgb_image = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
    rgb_image.flags.writeable = False  # 성능 최적화
    
    # 손 랜드마크 검출
    results = hands.process(rgb_image)
    
    # 결과 시각화를 위해 RGB→BGR 복원
    rgb_image.flags.writeable = True
    bgr_image = cv2.cvtColor(rgb_image, cv2.COLOR_RGB2BGR)
    
    # 검출된 손 랜드마크와 연결선 그리기
    if results.multi_hand_landmarks:
      for hand_landmarks in results.multi_hand_landmarks:
        mp_draw.draw_landmarks(
            bgr_image, hand_landmarks, mp_hands.HAND_CONNECTIONS)
    
    cv2.imshow('손 랜드마크 추적', bgr_image)
    if cv2.waitKey(1) & 0xFF == 27: break  # ESC 종료

cap.release()
cv2.destroyAllWindows()

MediaPipe - Face Mesh: 정밀 얼굴 랜드마크

3. 얼굴 메시 - 468개 정밀 포인트

Face Mesh는 얼굴에서 468개의 3D 랜드마크를 추출하여 입술, 눈, 눈썹, 얼굴 윤곽을 정밀하게 추적합니다. 표정 분석이나 AR 필터 제작에 필수적입니다.

정규화 좌표: 랜드마크는 0~1 범위로 화면 크기와 무관합니다. 실제 픽셀 좌표는 width/height를 곱해 계산합니다.

refine_landmarks=True: 눈동자와 입술 주변을 더 정밀하게 추적합니다.

💡 활용 팁

특정 랜드마크만 추출하려면 FACEMESH_LIPS, FACEMESH_LEFT_EYE 등을 사용하세요

import cv2, mediapipe as mp

# Face Mesh 솔루션 초기화
mp_face_mesh = mp.solutions.face_mesh
mp_draw = mp.solutions.drawing_utils

# 얼굴 메시 설정
face_mesh = mp_face_mesh.FaceMesh(
    max_num_faces=2,              # 최대 추적 얼굴 수
    refine_landmarks=True,       # 눈/입술 정밀도 향상
    min_detection_confidence=0.5,
    min_tracking_confidence=0.5
)

cap = cv2.VideoCapture(0)
while cap.isOpened():
    success, image = cap.read()
    if not success: break
    
    # BGR→RGB 변환 및 처리
    rgb_image = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)
    results = face_mesh.process(rgb_image)
    
    # 468개 랜드마크를 개별 점으로 그리기
    if results.multi_face_landmarks:
        for face_landmarks in results.multi_face_landmarks:
            for landmark in face_landmarks.landmark:
                # 정규화된 좌표를 픽셀 좌표로 변환
                x = int(landmark.x * image.shape[1])
                y = int(landmark.y * image.shape[0])
                
                # 작은 초록색 점으로 랜드마크 표시
                cv2.circle(image, (x, y), 1, (0, 255, 0), -1)
    
    cv2.imshow('얼굴 메시 - 468개 포인트', image)
    if cv2.waitKey(1) & 0xFF == 27: break

cap.release()
cv2.destroyAllWindows()

MediaPipe - Pose: 신체 랜드마크

4. 전신 자세 추적 - 33개 핵심 포인트

MediaPipe Pose는 머리부터 발끝까지 33개의 신체 랜드마크를 실시간으로 추적합니다. 운동 분석, 자세 교정, 피트니스 앱 개발에 활용됩니다.

POSE_CONNECTIONS: 어깨-팔꿈치-손목, 엉덩이-무릎-발목 등의 연결선을 자동으로 그려 골격을 시각화합니다.

성능 팁: 480p 해상도로 시작하여 필요시 720p로 올리거나, model_complexity=0으로 경량 모델을 사용하세요.

📐 주요 랜드마크

NOSE(0), LEFT_SHOULDER(11), RIGHT_SHOULDER(12), LEFT_HIP(23), RIGHT_HIP(24) 등이 핵심

import cv2, mediapipe as mp

MediaPipe - 간단 액션 인식: 손들기/앉기

5. 좌표 기반 간단 동작 인식

복잡한 AI 모델 없이도 랜드마크 좌표 비교로 기본 동작을 분류할 수 있습니다. 손 들기, 앉기/서기, 점프 등의 동작을 실시간으로 판별합니다.

좌표 비교 원리: 손목이 어깨보다 위에 있으면 손 든 상태, 엉덩이와 무릎의 y좌표 차이가 작으면 앉은 상태로 판단합니다.

확장 방법: 여러 프레임의 평균으로 노이즈 제거, 임계값 조정으로 민감도 최적화가 가능합니다.

🎯 실용 팁

visibility > 0.5인 랜드마크만 사용하여 가려진 부위로 인한 오류를 방지하세요

import cv2, mediapipe as mp

def detect_action(pose_landmarks):
    """랜드마크 좌표로 간단한 동작을 분류하는 함수"""
    landmarks = pose_landmarks.landmark
    
    # 주요 관절 랜드마크 가져오기
    left_shoulder = landmarks[mp.solutions.pose.PoseLandmark.LEFT_SHOULDER.value]
    left_wrist = landmarks[mp.solutions.pose.PoseLandmark.LEFT_WRIST.value]
    left_hip = landmarks[mp.solutions.pose.PoseLandmark.LEFT_HIP.value]
    left_knee = landmarks[mp.solutions.pose.PoseLandmark.LEFT_KNEE.value]
    
    # 동작 분류 로직
    if left_wrist.y < left_shoulder.y and left_wrist.visibility > 0.5:
        return "손 들기"
    elif abs(left_hip.y - left_knee.y) < 0.1:
        return "앉기"
    else:
        return "서 있기"

# 메인 실행 부분
mp_pose = mp.solutions.pose
mp_draw = mp.solutions.drawing_utils

with mp_pose.Pose(min_detection_confidence=0.5) as pose:
    cap = cv2.VideoCapture(0)
    
    while cap.isOpened():
        ret, frame = cap.read()
        if not ret: break
        
        # 자세 검출
        rgb_image = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
        results = pose.process(rgb_image)
        
        # 동작 인식 및 표시
        if results.pose_landmarks:
            action = detect_action(results.pose_landmarks)
            
            # 결과를 화면에 표시
            cv2.putText(frame, f"동작: {action}", (10, 30), 
                       cv2.FONT_HERSHEY_SIMPLEX, 1, (0, 255, 0), 2)
            
            # 골격도 그리기
            mp_draw.draw_landmarks(frame, results.pose_landmarks, mp_pose.POSE_CONNECTIONS)
        
        cv2.imshow('간단 동작 인식', frame)
        if cv2.waitKey(1) & 0xFF == 27: break

cap.release()
cv2.destroyAllWindows()

MediaPipe - 스쿼트 카운터: 각도 기반

9. 스쿼트 카운터 - 각도 계산 활용

3개 점(엉덩이-무릎-발목)으로 무릎 각도를 계산하여 스쿼트 동작을 자동 카운트합니다. 피트니스 앱이나 운동 분석에 직접 활용할 수 있습니다.

각도 임계값: 일반적으로 서 있을 때 160°, 스쿼트 최저점에서 90° 기준으로 up↔down 전환을 감지합니다.

상태 관리: "up"→"down"→"up" 순서로 완전한 1회 동작을 카운트하여 부분 동작을 제외합니다.

💪 개선 방법

양쪽 다리 각도의 평균 사용, 3-5프레임 평균으로 노이즈 제거, 속도 체크로 너무 빠른 동작 필터링

import numpy as np
import cv2, mediapipe as mp

def calculate_angle(point_a, point_b, point_c):
    """3개 점으로 각도를 계산하는 함수 (point_b가 꼭짓점)"""
    a = np.array([point_a.x, point_a.y])  # 첫 번째 점
    b = np.array([point_b.x, point_b.y])  # 꼭짓점 (무릎)
    c = np.array([point_c.x, point_c.y])  # 세 번째 점
    
    # 벡터 계산
    radians = np.arctan2(c[1] - b[1], c[0] - b[0]) - np.arctan2(a[1] - b[1], a[0] - b[0])
    angle = np.abs(radians * 180.0 / np.pi)
    
    if angle > 180.0:
        angle = 360 - angle
    
    return angle

# 스쿼트 카운터 메인 코드
mp_pose = mp.solutions.pose
mp_draw = mp.solutions.drawing_utils

counter = 0
stage = "up"  # "up" 또는 "down" 상태

with mp_pose.Pose(min_detection_confidence=0.5, min_tracking_confidence=0.5) as pose:
    cap = cv2.VideoCapture(0)
    
    while cap.isOpened():
        ret, frame = cap.read()
        if not ret: break
        
        # 자세 검출
        rgb_image = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
        results = pose.process(rgb_image)
        
        if results.pose_landmarks:
            landmarks = results.pose_landmarks.landmark
            
            # 왼쪽 다리 랜드마크 추출
            hip = landmarks[mp_pose.PoseLandmark.LEFT_HIP.value]
            knee = landmarks[mp_pose.PoseLandmark.LEFT_KNEE.value]
            ankle = landmarks[mp_pose.PoseLandmark.LEFT_ANKLE.value]
            
            # 무릎 각도 계산
            angle = calculate_angle(hip, knee, ankle)
            
            # 스쿼트 카운팅 로직
            if angle > 160:
                stage = "up"
            elif angle < 90 and stage == "up":
                stage = "down"
                counter += 1
            
            # 결과 표시
            cv2.putText(frame, f'카운트: {counter}', (10, 30), 
                       cv2.FONT_HERSHEY_SIMPLEX, 1, (255, 255, 255), 2)
            cv2.putText(frame, f'각도: {int(angle)}', (10, 70), 
                       cv2.FONT_HERSHEY_SIMPLEX, 1, (255, 255, 255), 2)
            
            # 자세 그리기
            mp_draw.draw_landmarks(frame, results.pose_landmarks, mp_pose.POSE_CONNECTIONS)
        
        cv2.imshow('스쿼트 카운터', frame)
        if cv2.waitKey(1) & 0xFF == 27: break

cap.release()
cv2.destroyAllWindows()

MediaPipe - Activity Tracker: 움직임/자세 기록

10. Activity Tracker - 움직임 패턴 분석

전신 랜드마크의 중심 좌표 변화를 시계열로 추적하여 사용자의 활동 패턴을 분석합니다. 앉기/서기/걷기 상태를 자동 분류할 수 있습니다.

움직임 측정: 프레임 간 중심점 거리를 계산하여 활동량을 수치화합니다.

패턴 인식: 움직임 히스토리에서 평균/최대값으로 앉음/서있음/걷기를 구분합니다.

📈 활용 아이디어

운동량 측정, 업무 자세 모니터링, 헬스케어 앱, 게임 제어, 재택근무 활동 추적

import cv2, mediapipe as mp
import numpy as np
from collections import deque
import json, time

class ActivityTracker:
    def __init__(self, history_size=300):
        """활동 추적기 초기화"""
        self.movement_history = deque(maxlen=history_size)  # 10초간 기록
        self.center_history = deque(maxlen=10)
        self.activity_log = []
        self.prev_center = None
        self.session_start = time.time()
    
    def calculate_body_center(self, landmarks):
        """전신 랜드마크에서 중심점 계산"""
        # 주요 랜드마크만 사용 (노이즈 제거)
        key_points = [
            landmarks[mp.solutions.pose.PoseLandmark.LEFT_SHOULDER.value],
            landmarks[mp.solutions.pose.PoseLandmark.RIGHT_SHOULDER.value],
            landmarks[mp.solutions.pose.PoseLandmark.LEFT_HIP.value],
            landmarks[mp.solutions.pose.PoseLandmark.RIGHT_HIP.value],
        ]
        
        x_coords = [p.x for p in key_points if p.visibility > 0.5]
        y_coords = [p.y for p in key_points if p.visibility > 0.5]
        
        if len(x_coords) >= 2:
            return (np.mean(x_coords), np.mean(y_coords))
        return None
    
    def update_activity(self, landmarks):
        """프레임마다 활동 상태 업데이트"""
        center = self.calculate_body_center(landmarks)
        if center is None:
            return None
        
        # 움직임 계산
        movement = 0
        if self.prev_center is not None:
            dx = center[0] - self.prev_center[0]
            dy = center[1] - self.prev_center[1]
            movement = np.hypot(dx, dy)
        
        self.movement_history.append(movement)
        self.center_history.append(center)
        self.prev_center = center
        
        # 활동 상태 분류
        if len(self.movement_history) >= 30:  # 1초간 데이터
            recent_movements = list(self.movement_history)[-30:]
            avg_movement = np.mean(recent_movements)
            max_movement = max(recent_movements)
            
            if avg_movement < 0.005:
                activity = "앉아있음"
            elif avg_movement < 0.02:
                activity = "서있음"
            else:
                activity = "움직임"
            
            return {
                'activity': activity,
                'movement_avg': avg_movement,
                'movement_max': max_movement,
                'timestamp': time.time() - self.session_start
            }
        
        return None
    
    def save_session(self, filename='activity_log.json'):
        """세션 데이터를 JSON으로 저장"""
        with open(filename, 'w', encoding='utf-8') as f:
            json.dump(self.activity_log, f, ensure_ascii=False, indent=2)

MediaPipe - Posture Checker: 어깨/목 정렬

11. Posture Checker - 실시간 자세 교정

어깨 기울기, 머리 전방 이동, 등의 굽힘 각도를 실시간으로 분석하여 올바른 자세 유지를 도와줍니다. 사무직 거북목 예방에 특히 유용합니다.

측정 항목: 어깨 수평도(±5°), 목-어깨 정렬도, 등 굽힘각(20° 이하 권장)

피드백: 경고음 + 시각적 가이드라인으로 즉시 교정 가능

💡 개선 방향

개인별 기준점 저장, 자세 히스토리 분석, 스트레칭 타이머, 모바일 알림 연동

import cv2, mediapipe as mp
import numpy as np
import math

class PostureChecker:
    def __init__(self):
        """자세 체커 초기화"""
        self.warnings = []
        self.good_posture_count = 0
        self.total_frames = 0
    
    def calculate_angle_2d(self, point1, point2):
        """두 점 사이의 각도 계산"""
        dx = point2[0] - point1[0]
        dy = point2[1] - point1[1]
        return math.degrees(math.atan2(dy, dx))
    
    def analyze_posture(self, landmarks):
        """자세 분석 및 피드백 생성"""
        self.warnings.clear()
        self.total_frames += 1
        
        # 주요 랜드마크 추출
        left_shoulder = landmarks[mp.solutions.pose.PoseLandmark.LEFT_SHOULDER.value]
        right_shoulder = landmarks[mp.solutions.pose.PoseLandmark.RIGHT_SHOULDER.value]
        nose = landmarks[mp.solutions.pose.PoseLandmark.NOSE.value]
        left_hip = landmarks[mp.solutions.pose.PoseLandmark.LEFT_HIP.value]
        right_hip = landmarks[mp.solutions.pose.PoseLandmark.RIGHT_HIP.value]
        
        # 1. 어깨 수평도 체크
        shoulder_y_diff = abs(left_shoulder.y - right_shoulder.y)
        if shoulder_y_diff > 0.05:  # 임계값 조정 가능
            side = "왼쪽" if left_shoulder.y < right_shoulder.y else "오른쪽"
            self.warnings.append(f"⚠️ {side} 어깨가 올라가 있습니다")
        
        # 2. 머리 전방 이동 체크 (거북목)
        shoulder_center_x = (left_shoulder.x + right_shoulder.x) / 2
        head_forward_distance = abs(nose.x - shoulder_center_x)
        if head_forward_distance > 0.1:
            self.warnings.append("🐢 목이 너무 앞으로 나와있습니다 (거북목)")
        
        # 3. 등 굽힘 각도 체크
        shoulder_center_y = (left_shoulder.y + right_shoulder.y) / 2
        hip_center_y = (left_hip.y + right_hip.y) / 2
        
        # 등의 기울기 계산 (간단한 근사)
        back_tilt = abs(shoulder_center_y - hip_center_y)
        if back_tilt < 0.3:  # 너무 앞으로 숙인 경우
            self.warnings.append("📐 등을 더 펴주세요")
        
        # 4. 양쪽 어깨 높이 차이로 좌우 균형 체크
        shoulder_tilt_angle = self.calculate_angle_2d(
            (left_shoulder.x, left_shoulder.y), 
            (right_shoulder.x, right_shoulder.y)
        )
        if abs(shoulder_tilt_angle) > 5:  # 5도 이상 기울어진 경우
            self.warnings.append(f"⚖️ 어깨 기울기: {shoulder_tilt_angle:.1f}° (±5° 이내 권장)")
        
        # 좋은 자세 카운트
        if len(self.warnings) == 0:
            self.good_posture_count += 1
            return "✅ 좋은 자세입니다!"
        
        return self.warnings
    
    def get_posture_score(self):
        """자세 점수 계산 (백분율)"""
        if self.total_frames == 0:
            return 0
        return int((self.good_posture_count / self.total_frames) * 100)

# 메인 실행 코드

mp_draw = mp.solutions.drawing_utils

posture_checker = PostureChecker()

with mp_pose.Pose(min_detection_confidence=0.5, min_tracking_confidence=0.5) as pose:
    cap = cv2.VideoCapture(0)
    
    while cap.isOpened():
        ret, frame = cap.read()
        if not ret: break
        
        # 자세 검출
        rgb_image = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
        results = pose.process(rgb_image)
        
        if results.pose_landmarks:
            # 자세 분석
            feedback = posture_checker.analyze_posture(results.pose_landmarks.landmark)
            score = posture_checker.get_posture_score()
            
            # 피드백 화면 출력
            y_offset = 30
            if isinstance(feedback, str):
                # 좋은 자세
                cv2.putText(frame, feedback, (10, y_offset), 
                           cv2.FONT_HERSHEY_SIMPLEX, 1, (0, 255, 0), 2)
            else:
                # 경고 메시지들
                for warning in feedback:
                    cv2.putText(frame, warning, (10, y_offset), 
                               cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 0, 255), 2)
                    y_offset += 25
            
            # 자세 점수 표시
            score_color = (0, 255, 0) if score >= 70 else (0, 165, 255) if score >= 50 else (0, 0, 255)
            cv2.putText(frame, f"자세 점수: {score}%", (10, frame.shape[0] - 20), 
                       cv2.FONT_HERSHEY_SIMPLEX, 0.7, score_color, 2)
            
            # 자세 스켈레톤 그리기
            mp_draw.draw_landmarks(frame, results.pose_landmarks, mp_pose.POSE_CONNECTIONS)
        
        cv2.imshow('실시간 자세 교정', frame)
        if cv2.waitKey(1) & 0xFF == 27: break

cap.release()
cv2.destroyAllWindows()
